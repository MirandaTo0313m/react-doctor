## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/architecture/no-lint-disable-comments.ts`:

```ts
import fs from "node:fs";
import { defineRule } from "../../utils/define-rule.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";

const LINT_DISABLE_COMMENT_PATTERN =
  /^(?:\/\/|\/\*|\*)\s*(?:eslint|oxlint)-disable(?:-(?:next-)?line)?\b/;

const hasLintDisableComment = (sourceText: string): boolean =>
  sourceText.split("\n").some((line) => LINT_DISABLE_COMMENT_PATTERN.test(line.trimStart()));

export const noLintDisableComments = defineRule<Rule>({
  id: "no-lint-disable-comments",
  severity: "warn",
  recommendation:
    "Fix the underlying lint violation instead of suppressing it inline. If the file is generated, vendored, or test-only, exclude it at the config boundary instead of adding an `eslint-disable` comment.",
  create: (context: RuleContext) => {
    const filename = context.getFilename?.();
    if (!filename) return {};

    let sourceText = "";
    try {
      sourceText = fs.readFileSync(filename, "utf8");
    } catch {
      return {};
    }

    if (!hasLintDisableComment(sourceText)) return {};

    return {
      Program(node: EsTreeNodeOfType<"Program">) {
        context.report({
          node,
          message:
            "Inline lint disable comment — remove the suppression and fix the lint issue directly",
        });
      },
    };
  },
});
```
