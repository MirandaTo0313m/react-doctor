## Fix prompt

Actionable fix suggestion surfaced to the user when the rule fires:

> Replace the indirect invocation with a clearer module load. If the dependency is known at build time, use a static import; if it must stay runtime-resolved, call `require()` directly instead of routing it through `.call(null, ...)`.

```ts
const mod = require(moduleName);
// or
import { mod } from "./mod";
```

## Positive fixture (SHOULD trigger)

## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/bundle-size/no-require-call.ts`:

```ts
import { defineRule } from "../../utils/define-rule.js";
import { isNodeOfType } from "../../utils/is-node-of-type.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";

const isNullLiteral = (node: unknown): boolean =>
  isNodeOfType(node, "Literal") && node.value === null;

export const noRequireCall = defineRule<Rule>({
  id: "no-require-call",
  severity: "warn",
  recommendation:
    "Use a static import, or call `require(moduleName)` directly if the module path must stay runtime-resolved",
  create: (context: RuleContext) => ({
    CallExpression(node: EsTreeNodeOfType<"CallExpression">) {
      if (!isNodeOfType(node.callee, "MemberExpression")) return;
      if (
        !isNodeOfType(node.callee.object, "Identifier") ||
        node.callee.object.name !== "require"
      ) {
        return;
      }
      if (
        !isNodeOfType(node.callee.property, "Identifier") ||
        node.callee.property.name !== "call"
      ) {
        return;
      }

      const thisArgument = node.arguments[0];
      if (!isNullLiteral(thisArgument)) return;

      context.report({
        node,
        message:
          "Indirect `require.call(null, ...)` hides the module load from bundlers — use a static import or a plain `require(...)` instead",
      });
    },
  }),
});
```
