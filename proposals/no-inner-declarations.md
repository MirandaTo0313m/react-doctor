## Validation prompt

FP-aware guidance for the [react-review agent](https://github.com/millionco/react-review) when triaging this rule:

> Verify the declaration is actually nested inside a control-flow or case block, not simply placed at the top of a module or directly inside a function body. Common false positives are exported helpers, direct function-body helpers, and `const foo = function () {}` patterns, which this rule should not flag. Only nested `if`, `for`, `while`, `try`, or `switch` placements should be treated as findings.

## Fix prompt

Actionable fix suggestion surfaced to the user when the rule fires:

> Hoist the helper to the surrounding scope, or rewrite it as a function expression if it must stay local. For example:

```ts
const getComponentTree = (): string => {
  /* ... */
};

if (enabled) {
  internalMcpFunctions.getComponentTree = getComponentTree;
}
```

## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/correctness/no-inner-declarations.ts`:

```ts
import { defineRule } from "../../utils/define-rule.js";
import { isNodeOfType } from "../../utils/is-node-of-type.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";

const isFunctionLikeNode = (node: unknown): boolean =>
  isNodeOfType(node, "FunctionDeclaration") ||
  isNodeOfType(node, "FunctionExpression") ||
  isNodeOfType(node, "ArrowFunctionExpression");

const isAllowedPlacement = (node: EsTreeNodeOfType<"FunctionDeclaration">): boolean => {
  const parent = node.parent;
  if (!parent) return false;
  if (isNodeOfType(parent, "Program")) return true;
  if (
    isNodeOfType(parent, "ExportNamedDeclaration") ||
    isNodeOfType(parent, "ExportDefaultDeclaration")
  ) {
    return true;
  }
  if (isNodeOfType(parent, "BlockStatement") && isFunctionLikeNode(parent.parent)) return true;
  return false;
};

export const noInnerDeclarations = defineRule<Rule>({
  id: "no-inner-declarations",
  severity: "warn",
  recommendation:
    "Hoist the helper to the surrounding scope, or rewrite it as a `const` function expression if it must stay local.",
  create: (context: RuleContext) => ({
    FunctionDeclaration(node: EsTreeNodeOfType<"FunctionDeclaration">) {
      if (isAllowedPlacement(node)) return;
      context.report({
        node,
        message:
          "Function declarations inside nested blocks can hoist unexpectedly; move this helper to the surrounding scope or use a function expression.",
      });
    },
  }),
});
```
