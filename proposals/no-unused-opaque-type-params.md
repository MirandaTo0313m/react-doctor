## Validation prompt

FP-aware guidance for the [react-review agent](https://github.com/millionco/react-review) when triaging this rule:

> Flag exported opaque types that keep a generic parameter but collapse the implementation to `mixed`, since that usually means the generic is only there to placate typecheck/linting. Do not flag phantom/brand types when the type parameter is actually threaded through the representation or upper bound, or when the generic is intentionally reserved and not part of the published API yet. Also skip non-exported local experiments and opaque types that have a concrete representation instead of `mixed`.

## Fix prompt

Actionable fix suggestion surfaced to the user when the rule fires:

> Drop the unused type parameter, or replace the placeholder `mixed` implementation with a representation that genuinely uses the generic. If the export only exists as a compatibility shim, make the type non-generic and keep the API surface explicit.

```js
// @flow
export opaque type ClientReference = mixed;
```

## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/correctness/no-unused-opaque-type-params.ts`:

```ts
import { defineRule } from "../../utils/define-rule.js";
import { isNodeOfType } from "../../utils/is-node-of-type.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";

const isPlaceholderOpaqueTypeExport = (
  node: EsTreeNodeOfType<"ExportNamedDeclaration">,
): boolean => {
  const declaration = node.declaration;
  if (!declaration || !isNodeOfType(declaration, "OpaqueType")) return false;
  if (!declaration.typeParameters) return false;
  if (declaration.supertype) return false;
  return isNodeOfType(declaration.impltype, "MixedTypeAnnotation");
};

export const noUnusedOpaqueTypeParams = defineRule<Rule>({
  id: "no-unused-opaque-type-params",
  severity: "warn",
  recommendation:
    "Remove the unused generic parameter from the exported opaque type, or give the opaque type a real generic representation instead of `mixed`.",
  create: (context: RuleContext) => ({
    ExportNamedDeclaration(node: EsTreeNodeOfType<"ExportNamedDeclaration">) {
      if (!isPlaceholderOpaqueTypeExport(node)) return;
      const declaration = node.declaration;
      if (!declaration || !isNodeOfType(declaration, "OpaqueType")) return;
      if (!isNodeOfType(declaration.id, "Identifier")) return;

      context.report({
        node: declaration.id,
        message: `Exported opaque type "${declaration.id.name}" has an unused generic parameter and is implemented as mixed`,
      });
    },
  }),
});
```
