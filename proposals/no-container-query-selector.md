## Validation prompt

FP-aware guidance for the [react-review agent](https://github.com/millionco/react-review) when triaging this rule:

> Flag this only when a test imports a testing library and then reaches into the RTL render `container` with `querySelector()` or `querySelectorAll()`. Common false positives are intentional low-level DOM assertions, SVG/portal/shadow-DOM inspection, or files where `container` is just an unrelated local variable. Do not flag tests that already use `within(...)` or another semantic query and only fall back to `container` for genuinely inaccessible markup.

## Fix prompt

Actionable fix suggestion surfaced to the user when the rule fires:

> Replace DOM traversal on the render container with an accessible query. If the element has a role or text, use `screen` or `within(...)` instead.

```tsx
const link = screen.getByRole("link", { name: /profile/i });
expect(link).toBeInTheDocument();
```

## Positive fixture (SHOULD trigger)

```tsx
import { render } from "@testing-library/react";

it("finds the link", () => {
  const { container } = render(<a href="/profile">Profile</a>);
  expect(container.querySelector("a")).not.toBeNull();
});
```

## Negative fixture (should NOT trigger)

```tsx
import { render, screen } from "@testing-library/react";

it("finds the link", () => {
  render(<a href="/profile">Profile</a>);
  expect(screen.getByRole("link", { name: "Profile" })).toBeInTheDocument();
});
```

## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/correctness/no-container-query-selector.ts`:

```ts
import { defineRule } from "../../utils/define-rule.js";
import { isTestLibraryImportSource } from "../../utils/is-test-library-import-source.js";
import { isNodeOfType } from "../../utils/is-node-of-type.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";

const QUERY_SELECTOR_METHOD_NAMES = new Set(["querySelector", "querySelectorAll"]);

export const noContainerQuerySelector = defineRule<Rule>({
  id: "no-container-query-selector",
  category: "correctness",
  severity: "warn",
  recommendation:
    "Use semantic React Testing Library queries such as `screen.getByRole(...)` or `within(...)` instead of `container.querySelector(...)`.",
  create: (context: RuleContext) => {
    let hasTestingLibraryImport = false;

    return {
      ImportDeclaration(node: EsTreeNodeOfType<"ImportDeclaration">) {
        if (hasTestingLibraryImport) return;
        if (isTestLibraryImportSource(node.source?.value)) {
          hasTestingLibraryImport = true;
        }
      },
      CallExpression(node: EsTreeNodeOfType<"CallExpression">) {
        if (!hasTestingLibraryImport) return;
        if (!isNodeOfType(node.callee, "MemberExpression")) return;
        if (!isNodeOfType(node.callee.object, "Identifier")) return;
        if (node.callee.object.name !== "container") return;
        if (!isNodeOfType(node.callee.property, "Identifier")) return;
        if (!QUERY_SELECTOR_METHOD_NAMES.has(node.callee.property.name)) return;

        context.report({
          node: node.callee.property,
          message:
            "Avoid `container.querySelector()` or `container.querySelectorAll()` in React Testing Library tests. Use `screen.getByRole(...)`, `screen.getByText(...)`, or `within(...)` instead.",
        });
      },
    };
  },
});
```
