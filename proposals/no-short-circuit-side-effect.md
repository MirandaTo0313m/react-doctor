## Validation prompt

FP-aware guidance for the [react-review agent](https://github.com/millionco/react-review) when triaging this rule:

> Check whether this is a standalone short-circuit expression used only to trigger a call, assignment, update, or `new` expression. Common false positives are test helpers, mocks, or deliberate guard clauses inside `if`/`while` conditions, plus JSX conditionals like `{flag && <Foo />}` which are not expression statements and should not be reported. Also ignore optional chaining calls and any logical expression whose value is genuinely being used.

## Fix prompt

Actionable fix suggestion surfaced to the user when the rule fires:

> Replace the short-circuit statement with an explicit `if` block, or use optional chaining when you're only invoking an optional callback. That makes the side effect obvious and avoids `no-unused-expressions` disables.

```ts
if (onReady) {
  onReady();
}
```

## Proposed AST detector

Would land at `packages/oxlint-plugin-react-doctor/src/plugin/rules/correctness/no-short-circuit-side-effect.ts`:

```ts
import { defineRule } from "../../utils/define-rule.js";
import type { EsTreeNode } from "../../utils/es-tree-node.js";
import type { EsTreeNodeOfType } from "../../utils/es-tree-node-of-type.js";
import { isNodeOfType } from "../../utils/is-node-of-type.js";
import type { Rule } from "../../utils/rule.js";
import type { RuleContext } from "../../utils/rule-context.js";
import { walkAst } from "../../utils/walk-ast.js";

const isFunctionLike = (node: EsTreeNode): boolean =>
  isNodeOfType(node, "ArrowFunctionExpression") ||
  isNodeOfType(node, "FunctionExpression") ||
  isNodeOfType(node, "FunctionDeclaration");

const hasSideEffectingExpression = (node: EsTreeNode): boolean => {
  let didFindSideEffect = false;

  walkAst(node, (child) => {
    if (child !== node && isFunctionLike(child)) return false;
    if (
      isNodeOfType(child, "CallExpression") ||
      isNodeOfType(child, "NewExpression") ||
      isNodeOfType(child, "AssignmentExpression") ||
      isNodeOfType(child, "UpdateExpression")
    ) {
      didFindSideEffect = true;
      return false;
    }
  });

  return didFindSideEffect;
};

export const noShortCircuitSideEffect = defineRule<Rule>({
  id: "no-short-circuit-side-effect",
  tags: ["test-noise"],
  severity: "warn",
  recommendation:
    "Replace the short-circuit statement with an explicit `if` block, or use optional chaining for a simple optional callback call.",
  create: (context: RuleContext) => ({
    ExpressionStatement(node: EsTreeNodeOfType<"ExpressionStatement">) {
      if (!isNodeOfType(node.expression, "LogicalExpression")) return;
      if (!(["&&", "||", "??"] as const).includes(node.expression.operator)) return;
      if (!hasSideEffectingExpression(node.expression)) return;

      context.report({
        node: node.expression,
        message:
          "Avoid using a short-circuit expression as a side-effect statement; use `if` or optional chaining instead.",
      });
    },
  }),
});
```
